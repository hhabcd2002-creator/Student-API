// Google Apps Script (For No/SerialNo/Mark format) function doGet(e) { var action = e.parameter.action; if (action == 'getData') { return ContentService.createTextOutput(JSON.stringify(getData())).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.clear(); // Headers အသစ် - No, Serial No, Mark ပုံစံဖြစ်ဖို့ sheet.appendRow(["StudentName", "No", "ExamSerialNo", "Marks"]); data.forEach(function(studentData) { studentData.exams.forEach(function(exam) { // ကျောင်းသားတစ်ယောက်ချင်းစီရဲ့ No, Serial No, Mark တွေကို Row တစ်ခုစီအနေနဲ့ထည့်မယ် sheet.appendRow([studentData.name, exam.no, exam.serial, exam.mark]); }); }); return ContentService.createTextOutput("Data Saved Successfully").setMimeType(ContentService.MimeType.TEXT); } function getData() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); var studentsMap = {}; for (var i = 1; i < rows.length; i++) { var name = rows[i][0]; // Student data ကို structure ပြန်ဖွဲ့ခြင်း if (!studentsMap[name]) { studentsMap[name] = { name: name, exams: [] }; } studentsMap[name].exams.push({ no: rows[i][1], serial: rows[i][2], mark: rows[i][3] }); } return Object.values(studentsMap); }
